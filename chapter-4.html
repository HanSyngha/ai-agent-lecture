함수 시그니처에
history 한 줄 추가.

def generate_response_with_history(user_prompt, history): messages = [{"role": "system", "content": system_prompt}] messages.extend(history) # 과거 대화 messages.extend([{"role": "user", "content": user_prompt}]) # 이번 턴 response = client.chat.completions.create( model=os.getenv("model"), messages=messages, stream=True, ) chunks = (c.choices[0].delta.content for c in response if c.choices) parts = [print(d, end="", flush=True) or d for d in chunks if d] return "".join(parts) history = [] # 시작은 빈 리스트

변한 곳은 messages 조립 부분뿐. system + history(N개) + 이번 user 순서.

Code · 매 턴의 마무리

매 턴 끝에
두 줄 추가.

# 1턴 user_prompt_1 = f"이력서 평가해줘 [이력서]{resume}" response_1 = generate_response_with_history(user_prompt_1, history) history.append({"role": "user", "content": user_prompt_1}) history.append({"role": "assistant", "content": response_1}) # 2턴 — history엔 이미 1턴 내용이 들어있음 user_prompt_2 = f"공고로 평가 [공고]{jp}" response_2 = generate_response_with_history(user_prompt_2, history) history.append({"role": "user", "content": user_prompt_2}) history.append({"role": "assistant", "content": response_2})

user/assistant 짝을 맞춰서 append. assistant 응답을 빼먹으면 LLM이 자기 답을 못 보고 다음 턴을 만듭니다.
